HTML Markup

This section contains HTML Markup to create layout empty. Default layout of the template comes with the same layout.

Vuexy has a ready to use starter kit, you can use this layout directly by using layout-empty.html. 


<!DOCTYPE html>
<html lang="en">
  <head>
    ...
  </head>
  <body class="vertical-layout vertical-menu-modern navbar-floating footer-static" data-menu="vertical-menu-modern">

    <!-- BEGIN: Navbar -->
    <nav class="header-navbar navbar navbar-expand-lg align-items-center floating-nav navbar-light navbar-shadow">
      ...
    </nav>
    <!-- END: Navbar -->

    <!-- BEGIN: Main Menu -->
    <div class="main-menu menu-fixed menu-light menu-accordion menu-shadow" data-scroll-to-active="true">
      ...
    </div>
    <!-- END: Main Menu -->

    <!-- BEGIN: Content -->
    <div class="app-content content">

        <!-- Content-wrapper -->
        <div class="content-wrapper">
          ...
        </div>

    </div>
    <!-- END: Content -->

    <!-- Begin: Footer -->
    <footer class="footer footer-static footer-light">
      ...
    </footer>
    <!-- END: Footer -->

  </body>
</html>
